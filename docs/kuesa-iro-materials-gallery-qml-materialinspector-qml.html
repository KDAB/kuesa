      visible: contentText.text !== ""

          opacity: 0.4
          color: "black"
          anchors.fill: parent
          radius: 15
          border {

      Text {
          color: "white"
          anchors.centerIn: parent
          id: contentText
          text: _materialInspector ? _materialInspector.description : ""
      }
