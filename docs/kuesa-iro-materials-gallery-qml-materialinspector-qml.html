 import QtQuick 2.12

 Item {
     readonly property int _margins: 25
     width: contentText.implicitWidth + _margins
     height: contentText.implicitHeight + _margins

     visible: contentText.text !== ""

     Rectangle {
         opacity: 0.4
         color: "black"
         anchors.fill: parent
         radius: 15
         border {
             color: "white"
             width: 1
         }
     }

     Text {
         color: "white"
         anchors.centerIn: parent
         id: contentText
         text: _materialInspector ? _materialInspector.description : ""
     }
 }
